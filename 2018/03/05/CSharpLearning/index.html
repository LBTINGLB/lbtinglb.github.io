public class Solution {
public IList> GroupAnagrams(string[] strs) {
if (strs.Length == 0) return strs;
//aabc -> #2#1#1, aac -> #2#0#1
Dictionary> d = new Dictionary>();
for(int i = 0; i < strs.Length; i++)
{
string s = strs[i];
int[] count = new int[26];
/ initialize elements of array n /
for (int i = 0; i < 26; i++ ) {
count[i] = 0;
}
// Get the frequency
foreach(char a in s)
{
count[a - ‘a’]++;
}
string key;
foreach(int value in count)
{
key +=”#”;
key += value.ToString();
}
if (d.ContainsKey(key))
{
d[key].Add(s);
}
else
{
IList value = new IList();
value.Add(s);
d[key] = value;
}
}
IList> answer = new IList>();
foreach(KeyValuePair> p in d)
{
answer.Add(p.Value);
}
return answer;
}
}
